create TABLE SUPPLIER(SUPPLIERID NUMBER(4) PRIMARY KEY,
  2  SNAME VARCHAR(30),
  3  SCITY VARCHAR(30),
  4  SPHONE NUMBER(10),
  5  EMAIL VARCHAR(40));

 CREATE TABLE ITEMS(ITEMID NUMBER(4) PRIMARY KEY,
  2  ITEMNAME VARCHAR(35) NOT NULL,
  3  SUPPLIERID NUMBER(4),
  4  MINQTY NUMBER(4) NOT NULL,
  5  MAXQTY NUMBER(4) NOT NULL,
  6  PRICE NUMBER(5,2),
  7  FOREIGN KEY(SUPPLIERID) REFERENCES SUPPLIER(SUPPLIERID));

 CREATE TABLE ORDERS(ORDERID NUMBER(4) PRIMARY KEY,
  2  ORDERDATE DATE,
  3  CUSTID NUMBER(4),
  4  QUANTITY NUMBER(4) CHECK(QUANTITY > 0),
  5  ITEMID NUMBER(4),
  6  FOREIGN KEY(ITEMID) REFERENCES ITEMS(ITEMID));

CREATE TABLE CUSTOMER(CUSTID NUMBER(4) PRIMARY KEY,
  2  FNAME CHAR(30),
  3  LNAME CHAR(30),
  4  ADDRESS CHAR(30),
  5  PHONENO NUMBER(10) NOT NULL,
  6  CITY CHAR(30),
  7  COUNTRY CHAR(30),
  8  DATEFIRSTPURCHASED DATE,
  9  SUPPLIERID NUMBER(4),
 10  FOREIGN KEY(SUPPLIERID) REFERENCES SUPPLIER(SUPPLIERID));
